众数问题（分治法）

Aug 25, 2022
阅读时长: 2 分钟

问题描述

给定含有n个元素的多重集合S，每个元素在S中出现的次数成为该元素的重数。多重集S中重数最大的元素成为众数。
例如，S={1,2,2,2,3,3,5}。多重集S的众数为2，其重数为3.

编程任务

对于给定的由n个自然数组成的多重集S， 计算S的众数及其重数。如果出现多个众数，请输出最小的那个。

解题思路

  1. 先对S进行排序，目的将相同的数聚集在一起，方便后续操作。

  2. 选定数组中间的数作为界限，并以此将数组分为两部分。

  3. 计算中间部分的众数及其众数。

  4. 左右部分的众数及其重数，回到步骤2。

  5. 比较三部分的众数，并选择符合题意的输出。

#include <stdio.h>
#include <stdlib.h>

void swap(int &a, int &b)
{
    int temp = a;
    a = b;
    b = temp;
}//交换两个数

int partition(int num[], int l, int r)
{
    int i = l;
    int j = r+1;
    int key = num[l];
    while (true)
    {
        while (num[++i] <= key);
        while (num[--j] > key);
        if (i > j)
        {
            break;
        }
        swap(num[i], num[j]);        
    }
    num[l] = num[j];
    num[j] = key;
    return j;
}

void quicksort(int num[], int l, int r)
{
    if (r > l)
    {
        int p = partition(num, l, r);
        quicksort(num, l, p-1);
        quicksort(num, p+1, r);
    }
}//快速排序

void fun(int num[], int l, int r, int &mode, int &modeNum)
{
    int p = (l+r)/2;

    int lmax = p;
    while (num[--lmax] == num[p]);

    int rmin = p;
    while (num[++rmin] == num[p]);
    //切割数组

    mode = num[p];
    modeNum = rmin-lmax-1;
    //将中间部分的众数暂定为输出结果

    int lmode = 0;
    int lmodeNum = 0;//左众数、重数
    if (lmax-l+1 >= modeNum)
    {
        fun(num, l, lmax, lmode, lmodeNum);
    }//当左半集合元素个数大于等于modeNum时才有必要考虑左半部分

    if (lmodeNum > modeNum)
    {
        mode = lmode;
        modeNum = lmodeNum;
    }
    else if (lmodeNum == modeNum)
    {
        if (lmode < mode)
        {
            mode = lmode;
        }        
    }//左众数与当前众数比较，并替换


    int rmode = 0;
    int rmodeNum = 0;//右众数、重数
    if (r-rmin+1 >= modeNum)
    {
        fun(num, rmin, r, rmode, rmodeNum);
    }//当右半集合元素个数大于等于modeNum时才有必要考虑右半部分
    
    if (rmodeNum > modeNum)
    {
        mode = rmode;
        modeNum = rmodeNum;
    }
    else if (rmodeNum == modeNum)
    {
        if (rmode < mode)
        {
            mode = rmode;
        }        
    }//右众数与当前众数比较，并替换
    
}

int main()
{
    int num[28]={2,2,3,3,3,1,1,1,1,6,6,6,4,4,4,5,5,2,2,3,4,4,6,6,3,5,5,5};
    quicksort(num, 0, 28);

    int mode, modeNum;
    fun(num, 0, 28, mode, modeNum);
    printf("%d, %d", mode, modeNum);
    system("pause");
    return 0;
}
